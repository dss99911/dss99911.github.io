- **마커 클릭**: 식당 상세 정보 (별점, 리뷰수, 저장수, 가격대, 주소 등) 확인

    val = Math.sqrt(parseNum(r.save)) / Math.sqrt(maxSave);
  } else if (mode === 'score') {
    val = r.score && maxScore > minScore ? (r.score - minScore) / (maxScore - minScore) : 0.5;
  } else {
    // composite
    const rv = Math.sqrt(parseNum(r.review)) / Math.sqrt(maxReview);
    const sv = Math.sqrt(parseNum(r.save)) / Math.sqrt(maxSave);
    const sc = r.score && maxScore > minScore ? (r.score - minScore) / (maxScore - minScore) : 0.5;
    val = sc * 0.3 + rv * 0.35 + sv * 0.35;
  }
  return 5 + val * 30; // 5~35px, sqrt로 수치 비례
}

function getOpacity(r) {
  const rv = Math.sqrt(parseNum(r.review)) / Math.sqrt(maxReview);
  const sv = Math.sqrt(parseNum(r.save)) / Math.sqrt(maxSave);
  return 0.4 + (rv * 0.3 + sv * 0.3);
}

function renderMarkers() {
  markers.forEach(m => map.removeLayer(m));
  markers = [];
  let count = 0;

  RESTAURANTS.forEach(r => {
    const group = getCatGroup(r.cat);
    if (disabledGroups.has(group)) return;

    const cfg = CAT_GROUPS[group];
    const radius = getRadius(r, sizeMode);
    const opacity = getOpacity(r);

    const marker = L.circleMarker([parseFloat(r.y), parseFloat(r.x)], {
      radius: radius,
      fillColor: cfg.color,
      color: cfg.color,
      weight: 1.5,
      opacity: 0.9,
      fillOpacity: opacity * 0.65
    }).addTo(map);

    marker.bindPopup(createPopup(r), { maxWidth: 300 });
    marker.bindTooltip(r.name, { direction: 'top', offset: [0, -radius] });
    markers.push(marker);
    count++;
  });

  document.getElementById('stats').textContent = `표시 중: ${count} / ${RESTAURANTS.length}개`;
}

// Legend
function renderLegend() {
  const el = document.getElementById('legend');
  const groupCounts = {};
  RESTAURANTS.forEach(r => {
    const g = getCatGroup(r.cat);
    groupCounts[g] = (groupCounts[g] || 0) + 1;
  });

  el.innerHTML = Object.entries(CAT_GROUPS)
    .filter(([g]) => groupCounts[g])
    .sort((a, b) => (groupCounts[b[0]] || 0) - (groupCounts[a[0]] || 0))
    .map(([group, cfg]) => {
      const disabled = disabledGroups.has(group) ? ' disabled' : '';
      return `<span class="legend-item${disabled}" data-group="${group}" style="background:${cfg.color}11;border-color:${cfg.color}44">
        <span class="legend-dot" style="background:${cfg.color}"></span>
        ${cfg.emoji} ${group} <span style="color:#999">${groupCounts[group]}</span>
      </span>`;
    }).join('');

  el.querySelectorAll('.legend-item').forEach(item => {
    item.addEventListener('click', () => {
      const g = item.dataset.group;
      if (disabledGroups.has(g)) disabledGroups.delete(g);
      else disabledGroups.add(g);
      renderLegend();
      renderMarkers();
    });
  });
}

const LocControl = L.Control.extend({
  options: { position: 'topleft' },
  onAdd() {
    const btn = L.DomUtil.create('div', 'loc-btn');
    btn.innerHTML = '\uD83D\uDCCD';
    btn.title = '내 위치';
    L.DomEvent.disableClickPropagation(btn);
    btn.addEventListener('click', () => locateMe(btn));
    return btn;
  }
});
map.addControl(new LocControl());

let myLocMarker = null;
let myLocRing = null;

function locateMe(btn) {
  if (!navigator.geolocation) { alert('이 브라우저에서 위치 서비스를 지원하지 않습니다.'); return; }
  btn.classList.add('loading');
  navigator.geolocation.getCurrentPosition(
    (pos) => {
      btn.classList.remove('loading');
      const lat = pos.coords.latitude;
      const lng = pos.coords.longitude;
      const acc = pos.coords.accuracy;
      if (myLocMarker) { map.removeLayer(myLocMarker); map.removeLayer(myLocRing); }
      myLocMarker = L.marker([lat, lng], {
        icon: L.divIcon({ className: 'my-loc-marker', iconSize: [16, 16], iconAnchor: [8, 8] }),
        zIndexOffset: 1000
      }).addTo(map).bindPopup('현재 내 위치');
      myLocRing = L.circle([lat, lng], { radius: acc, className: 'my-loc-ring', interactive: false }).addTo(map);
      map.setView([lat, lng], 14);
    },
    (err) => {
      btn.classList.remove('loading');
      if (err.code === 1) alert('위치 권한이 거부되었습니다. 브라우저 설정에서 위치 권한을 허용해주세요.');
      else alert('위치를 가져올 수 없습니다: ' + err.message);
    },
    { enableHighAccuracy: true, timeout: 10000 }
  );
}
