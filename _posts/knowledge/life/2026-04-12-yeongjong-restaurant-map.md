function getCatGroup(cat) {
  const mainCat = (cat || '').split(',')[0].trim();
  for (const [group, cfg] of Object.entries(CAT_GROUPS)) {
    if (cfg.match.some(m => mainCat.includes(m))) return group;
  }
  return '기타';
}

function parseNum(s) {
  if (!s) return 0;
  return parseInt(String(s).replace(/[,+~]/g, '')) || 0;
}

// Compute normalized scores
const reviews = RESTAURANTS.map(r => parseNum(r.review));
const saves = RESTAURANTS.map(r => parseNum(r.save));
const scores = RESTAURANTS.map(r => r.score || 0);
const maxReview = Math.max(...reviews);
const maxSave = Math.max(...saves);
const minScore = Math.min(...scores);
const maxScore = Math.max(...scores);

function getRadius(r, mode) {
  let val;
  if (mode === 'review') {
    val = Math.log1p(parseNum(r.review)) / Math.log1p(maxReview);
  } else if (mode === 'save') {
    val = Math.log1p(parseNum(r.save)) / Math.log1p(maxSave);
  } else if (mode === 'score') {
    val = maxScore > minScore ? (r.score - minScore) / (maxScore - minScore) : 0.5;
  } else {
    // composite
    const rv = Math.log1p(parseNum(r.review)) / Math.log1p(maxReview);
    const sv = Math.log1p(parseNum(r.save)) / Math.log1p(maxSave);
    const sc = maxScore > minScore ? (r.score - minScore) / (maxScore - minScore) : 0.5;
    val = sc * 0.3 + rv * 0.35 + sv * 0.35;
  }
  return 6 + val * 22; // 6~28px
}

function getOpacity(r) {
  const rv = Math.log1p(parseNum(r.review)) / Math.log1p(maxReview);
  const sv = Math.log1p(parseNum(r.save)) / Math.log1p(maxSave);
  return 0.5 + (rv * 0.3 + sv * 0.2);
}

function renderMarkers() {
  markers.forEach(m => map.removeLayer(m));
  markers = [];
  let count = 0;

  RESTAURANTS.forEach(r => {
    const group = getCatGroup(r.cat);
    if (disabledGroups.has(group)) return;

    const cfg = CAT_GROUPS[group];
    const radius = getRadius(r, sizeMode);
    const opacity = getOpacity(r);

    const marker = L.circleMarker([parseFloat(r.y), parseFloat(r.x)], {
      radius: radius,
      fillColor: cfg.color,
      color: cfg.color,
      weight: 1.5,
      opacity: 0.9,
      fillOpacity: opacity * 0.65
    }).addTo(map);

    marker.bindPopup(createPopup(r), { maxWidth: 300 });
    marker.bindTooltip(r.name, { direction: 'top', offset: [0, -radius] });
    markers.push(marker);
    count++;
  });

  document.getElementById('stats').textContent = `표시 중: ${count} / ${RESTAURANTS.length}개`;
}

// Legend
function renderLegend() {
  const el = document.getElementById('legend');
  const groupCounts = {};
  RESTAURANTS.forEach(r => {
    const g = getCatGroup(r.cat);
    groupCounts[g] = (groupCounts[g] || 0) + 1;
  });

  el.innerHTML = Object.entries(CAT_GROUPS)
    .filter(([g]) => groupCounts[g])
    .sort((a, b) => (groupCounts[b[0]] || 0) - (groupCounts[a[0]] || 0))
    .map(([group, cfg]) => {
      const disabled = disabledGroups.has(group) ? ' disabled' : '';
      return `<span class="legend-item${disabled}" data-group="${group}" style="background:${cfg.color}11;border-color:${cfg.color}44">
        <span class="legend-dot" style="background:${cfg.color}"></span>
        ${cfg.emoji} ${group} <span style="color:#999">${groupCounts[group]}</span>
      </span>`;
    }).join('');

  el.querySelectorAll('.legend-item').forEach(item => {
    item.addEventListener('click', () => {
      const g = item.dataset.group;
      if (disabledGroups.has(g)) disabledGroups.delete(g);
      else disabledGroups.add(g);
      renderLegend();
      renderMarkers();
    });
  });
}
